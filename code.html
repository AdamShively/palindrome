                        //Get the user input values.
                        function getValues() {

                            //Get start value that the user entered and parse it to an integer.
                            let start = document.getElementById("start").value;
                            start = parseInt(start);

                            //Get end value that the user entered and parse it to an integer.
                            let end = document.getElementById("end").value;
                            end = parseInt(end);

                            //Make sure user entered integers.
                            let isNumber = Number.isInteger(start) && Number.isInteger(end)

                            //Make sure user entries are within range 0-100 (inclusive).
                            let isPositive = start >=0 && end >=0 && end <= 100;

                            //Make sure user entered a valid range.
                            let isValidRange = start < end;

                            //Make sure user entries are valid.
                            if (isNumber && isPositive && isValidRange) {

                                let numbers = generateValues(start, end);
                                displayValues(numbers);
                            }
                            else {

                                //Show the alert box.
                                alert("Please make sure your entries are valid.");
                            }
                        }

                        //Generate values within range entered by the user.
                        function generateValues(start, end) {

                            let numbers= []

                            //Push all numbers (inclusive) from start to end.
                            for (let i = start; i <= end; i++) {

                                //Push all numbers to array.
                                numbers.push(i);
                            }

                            return numbers;
                        }

                        //Display values within range entered by the user.
                        function displayValues(numbers) {
                            
                            let templateValues = "";
                            
                            for (let i = 0; i < numbers.length; i++) {

                                let number = numbers[i];
                                let cssClass = "odd";

                                //Number is a multiple of ten.
                                if (number % 10 == 0) {
                                    cssClass = "even multiple-of-ten";
                                }
                                //Number is even.
                                else if (number % 2 == 0) {
                                    cssClass = "even";
                                }

                                //Add template literals to string.
                                //Prism does not properly display template literals,
                                //refer to git repo to view.
                                templateValues += `${number}`;
                            }
                            //Write values to the page.
                            document.getElementById("results").innerHTML = templateValues;
                        }

The Javascript Functions

The getValues Function:

This function gets the user's entries and initializes them to "start" and "end" then parses them to integers. Entries are then checked for validity using "isNumber", "isPositive", and "isValidRange". If the entries are valid then a function call of "generateValues" with "start" and "end" as arguements is made and initialized to "numbers". The "displayValues" function is called with "numbers" passed as its arguement. If the user's entries were invalid, an alert box is displayed telling the user to make sure the entries are valid.

The generateValues Function:

This function generates values within range entered by the user and pushes them to an array. It initializes the "numbers" array, loops through values within range entered by the user and pushes them to that array, and then returns that array.

The displayValues Function:

This function displays the values within range entered by the user. The string "templateValues" initialized and is used to collect values to be displayed. The parameter "numbers" is looped through and its elements are intialized to "number" and "cssClass" is initalized with each loop. If "number" is a multiple of ten, "cssClass" is set to "even multiple-of-ten". If "number" is even, "cssClass" is set to "even". Otherwise, "cssClass" remains set to "odd". These values are added to "templateValues". Follwoing the loop, "templateValues" is written to the page.
