                        //Get the user's input.
                        function getValue() {

                            //Get input that the user entered.
                            let input = document.getElementById("input").value;
                            
                            //Remove any invalid characters using regex.
                            let regex = /[^a-z0-9]/gi;
                            input = input.replace(regex, "");

                            //Make sure user entered input that can be checked.
                            if (input.length > 0) {

                                //Check if palindrome and get reverse.
                                let info = isPalindrome(input);

                                //Call function to display info.
                                displayValue(info);
                            }
                            else {
                                //Show the alert box.
                                alert("Please input a word, phrase, or sequence to check.");
                            }
                        }

                        //Check if input is a palidrome.
                        function isPalindrome(input) {

                            //Length of the user's input.
                            let inputLen = input.length;

                            if (inputLen == 1) {

                                //Return info in a dictionary.
                                return {isPal: true, reverse: input};
                            }

                            let reverse = "";

                            //Loop through characters and add to a new string.
                            for (let i = input.length-1; i >= 0; i--) {

                                reverse += input[i];
                            }

                            //Determine if user input is a palindrome.
                            let isPal = input.toLowerCase() == reverse.toLowerCase();

                            //Return info in a dictionary.
                            return {isPal: isPal, reverse: reverse};
                        }

                        //Display the results.
                        function displayValue(info) {

                            //Get values from dictionary.
                            let isPal = info.isPal;
                            let reverse = info.reverse;

                            //Use ternary operatory to initialize message.
                            message = isPal ? "Your entry is a Palindrome!" : 
                                            "Your entry is NOT a Palindrome!";

                            //Write to the page.
                            document.getElementById("ispal").innerHTML = message;
                            document.getElementById("info").innerHTML = `The reverse 
                                                        of your entry is: ${reverse}`;

                            //Allow display message to be displayed.
                            document.getElementById("alert").classList.remove("invisible");
                        }

The Javascript Functions

The getValues Function:

This function gets the user's entries and initializes them to "start" and "end" then parses them to integers. Entries are then checked for validity using "isNumber", "isPositive", and "isValidRange". If the entries are valid then a function call of "generateValues" with "start" and "end" as arguements is made and initialized to "numbers". The "displayValues" function is called with "numbers" passed as its arguement. If the user's entries were invalid, an alert box is displayed telling the user to make sure the entries are valid.

The generateValues Function:

This function generates values within range entered by the user and pushes them to an array. It initializes the "numbers" array, loops through values within range entered by the user and pushes them to that array, and then returns that array.

The displayValues Function:

This function displays the values within range entered by the user. The string "templateValues" initialized and is used to collect values to be displayed. The parameter "numbers" is looped through and its elements are intialized to "number" and "cssClass" is initalized with each loop. If "number" is a multiple of ten, "cssClass" is set to "even multiple-of-ten". If "number" is even, "cssClass" is set to "even". Otherwise, "cssClass" remains set to "odd". These values are added to "templateValues". Follwoing the loop, "templateValues" is written to the page.
